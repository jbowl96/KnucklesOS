33 * This function is use to issue system requests

35 *
36 * Parameters: op_code: Requested Operation, one of
37 * READ, WRITE, IDLE, EXIT
38 * device_id: For READ & WRITE this is the
39 * device to which the request is
40 * sent. One of DEFAULT_DEVICE or

42 * buffer_ptr: pointer to a character buffer
43 * to be used with READ & WRITE request
44 * count_ptr: pointer to an integer variable
45 * containing the number of characters

47 *
48 *************************************************/

54 {

57  if (op_code == IDLE || op_code == EXIT){

62  }// idle or exit
63 
64  else if (op_code == READ || op_code == WRITE)
65 {
66  // validate buffer pointer and count pointer

71 
72 // if parameters are valid store in the params structure
73  if ( return_code == 0){
74  params.op_code = op_code;
